import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Data Collection
# Import csv table from https://www.kaggle.com/aayushmishra1512/twitchdata
twitch_table = pd.read_csv("twitchdata-update.csv")

# Display a preview of some of the data
twitch_table = twitch_table.sort_values(by = ['Stream time(minutes)','Followers gained','Views gained'], ascending = False)
twitch_table = twitch_table.drop(['Partnered', 'Mature'], axis=1)
twitch_table = twitch_table.reset_index(drop = True)
